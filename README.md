```
...

    ...
    <repository>
      <id>com.blueskiron-public-repo RELEASES</id>
      <name>com.blueskiron maven public</name>

      </releases>

      </snapshots>
      <url>https://raw.github.com/jurajzachar/com.blueskiron-public-repo.git/releases</url>
    </repository>
    ...
 </repositories>
...
```
